### Glossário de Investimentos
# Ref: https://zeroheight.com/33af959d2/v/latest/p/803efb-glossario
# Termos que NÃO devem ser usados como sinônimos entre si:
- "desempenho" ≠ "valorização" ≠ "evolução" ≠ "variação"
- "rendimento" ≠ "rentabilidade" ≠ "remuneração"
- "dividendos" ≠ "proventos"
- "liquidação" ≠ "liquidez"
- "cota" ≠ "cotização"
- "composição" ≠ "distribuição"

# Termos com caixa-alta obrigatória:
- "Banco Central" / "BC" / "BCB" / "Bacen"
- "Bolsa de Valores"
- "Certificados de Depósito Interbancário" / "CDI"
- "Day Trade"  # sempre em itálico
- "Fundo Garantidor de Créditos" / "FGC"
- "Índice da Bolsa de Valores de São Paulo" / "Ibovespa" / "Ibov."
- "Previdência Privada" / "Prev. Privada"  # não usar "Previdência Priv."
- "Sistema Especial de Liquidação e de Custódia" / "Selic" / "Taxa Selic"
- "Tesouro Direto"  # não usar apenas "Tesouro"

### Evitar (termos proibidos ou desaconselhados)
# Ref: https://zeroheight.com/33af959d2/v/latest/p/03a147-expressoes-nao-recomendadas
- "banco 1" / "banco 2" / "banco X"  # placeholders genéricos. Substitua por "Bradesco" ou "Banco Bradesco"
- "Adquirir"  # usar "contratar" ou "comprar"
- "Através"  # usar "por meio de", "pelo", "pela"
- "Deixar na mão" / "Ficar na mão"  # conotação sexual
- "Efetuar" / "Realizar"  # usar "fazer" ou "concluir"
- "Obter"  # usar "ter"
- "Utilizar"  # usar "usar" (exceção: contexto de limites)
- "Aposta" / "Chance"  # apenas no contexto de investimentos; remetem à sorte

# Ref: https://zeroheight.com/33af959d2/v/latest/p/10d1a4-expressoes-informais
### Interjeições padronizadas
- "Ah" -> ["Ei", "Ué"]  # frustração ou lembrete
- "Humm" -> ["Hmm", "Hum", "Hã", "Ham"]  # dúvida/espera; sempre com dois "m"
- "Oba" -> ["Eba", "Uhul", "Yay"]  # entusiasmo
- "Ok" -> ["Beleza", "Tá bom"]  # concordância
- "Ops" -> ["Opa", "Eita", "Ih"]  # surpresa diante de erro
- "Poxa" -> ["Nossa", "Caramba", "Que pena"]  # frustração

### Contrações e reduções permitidas
# Ref: https://zeroheight.com/33af959d2/v/latest/p/10d1a4-expressoes-informais
- "pra" -> []  # permitido em ambientes transacional e institucional; NÃO usar "pro", "pras", "pros"
- "num" / "numa" -> []  # permitido com restrição de caracteres e tom informal; NÃO usar em erros, alertas ou valores
- "tá" -> []  # permitido apenas em benefícios e promoções; NÃO usar em mensagens de erro
